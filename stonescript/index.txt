func IsBlade(count)
  ?getCool(＂blade＂) <= 0 & 
  ^item.CanActivate() & 
  ^foe.count >= count
    screen.next()
    return foe.distance < 10
  :
    return false
func Blade()
  equip blade
  activate R
  next = false
  screen.resetOffset()
 
func useBlade()
 ?allowBlade
  ?getCool(＂blade＂) <= 0
   ?foe.distance <= 30
    equip blade
    activate R
    allowBlade = false
    next = false
    screen.resetOffset()

func IsBardiche(a)
 ?(getCool(＂bardiche＂) > 870 | 
  ^getCool(＂bardiche＂) <= 0) &
  ^item.CanActivate() & 
  ^foe.distance < 10 & 
  ^foe.hp+foe.armor >= a
  return true
 :
  return false

func Bardiche()
 equip Bardiche
 activate R

func IsQuart()
  ?getCool(＂quarterstaff＂) <= 0 &
  ^item.CanActivate()
    return True
  :
    return False

func Quart()
  equip quart *0
  activate R

func EnoughDistance()
  ?foe.distance <= 17 & foe.distance >= 11
    return true
  :
    return false

func IsDash()
  ?getCool(＂dash＂) <= 0 & EnoughDistance()
    return true
  :
    return false

func IsBash()
  ?getCool(＂bash＂) <= 0 & EnoughDistance()
    return true
  :
    return false

func IsHammer()
  ?(getCool(＂hammer＂) <= 0 |
  ^getCool(＂hammer＂) > 646) &
  ^item.CanActivate() &
  ^foe.distance < 21
    return true
  :
    return false

func Hammer()
  equip heavy hammer
  activate R

func SkeletonR()
  ?item.GetCooldown(＂skeleton_arm＂) <= 0 & 
  ^foe.hp + foe.armor <= 32
    equip skeleton arm
    ?item.CanActivate(＂skeleton_arm＂)
      activate R
    return true
  :
    return false

func IsMask(a)
  ?(getCool(＂mask＂) <= 0 ) &
  ^item.CanActivate() &
  ^buffs.count > 20 & 
  ^foe.hp+foe.armor >= a
    return true
  :
    return false

func Mask()
  equipR mask
  activate R

func BFbuffCount()
  var smiteCount = 0
  ?buffs.count > 0
    ?string.IndexOf(buffs.string,smite) > 0
      smiteCount = string.Sub(buffs.string, 
      ^string.IndexOf(buffs.string,smite) + 6,4)
      smiteCount = string.Sub(smiteCount, 0, 
      ^string.IndexOf(smiteCount,＂:＂))
      smiteCount = int.parse(smiteCount)
  return smiteCount
  
func SKAbuffCount()
  var pickCount = 0
  ?buffs.count > 0
    var pickCount = 0
    ?string.IndexOf(buffs.string,pick_pocket)
      ^ > 0
      pickCount = string.Sub(buffs.string, 
      ^string.IndexOf(buffs.string,
      ^pick_pocket) + 12, 3)
      pickCount = string.Sub(pickCount, 0, 
      ^string.IndexOf(pickCount,＂:＂))
      pickCount = int.parse(pickCount)
  return pickCount
